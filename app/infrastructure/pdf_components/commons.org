def crear_tabla_ficha(data_filas, col_widths, row_heights):
    """Función maestra para dar el look 'aireado' a cualquier ficha"""
    tabla = Table(data_filas, colWidths=col_widths, rowHeights=row_heights)
    
    # Estilo base que ya definimos
    estilo = [
        ('VALIGN', (0, 0), (-1, -1), 'MIDDLE'),
        ('LEFTPADDING', (0, 0), (-1, -1), 6),
        ('FONTNAME', (0, 0), (-1, -1), 'Helvetica'),
        ('FONTSIZE', (0, 0), (-1, -1), 9),
    ]
    
    # Aplicar gris intercalado automáticamente a las celdas de datos
    # (Esto se puede sofisticar, pero por ahora lo dejamos manual por ficha)
    tabla.setStyle(TableStyle(estilo))
    return tabla

  def draw_local_image_absolute(canvas, file_path, x_cm, y_cm, width_cm, height_cm):
      """
      Lee una imagen directamente desde el disco local y la dibuja en una 
      posición absoluta usando coordenadas FPDF-style (desde arriba hacia abajo).
      """
      try:
          # 1. Validar existencia del archivo
          if not os.path.exists(file_path):
              print(f"ERROR: El archivo no existe en la ruta: {file_path}")
              return

          # 2. Verificar que no esté vacío
          if os.path.getsize(file_path) < 100:
              print(f"ADVERTENCIA: El archivo {file_path} parece estar corrupto o vacío.")
              return


          # 3. Preparar el lector de imagen (ReportLab maneja rutas de archivo directamente)
          # Usamos ImageReader para mayor compatibilidad, similar a tu versión URL
          img_reader = ImageReader(file_path)

          # 4. CONVERSIÓN DE COORDENADAS (El truco maestro de Cerebro)
          # ReportLab dibuja desde abajo, FPDF desde arriba.
          page_height = 29.7 * cm
          y_reportlab = page_height - (y_cm * cm) - (height_cm * cm)

          # 5. Dibujar en el canvas
          canvas.drawImage(
              img_reader, 
              x_cm * cm, 
              y_reportlab,
              width=width_cm * cm, 
              height=height_cm * cm,
              mask='auto'
          )

          print(f"Imagen local dibujada con éxito: {os.path.basename(file_path)} en X:{x_cm} Y:{y_cm}")

      except Exception as e:
          print(f"ERROR CRÍTICO al procesar imagen local: {str(e)}")

def draw_url_image_absolute(canvas, url, x_cm, y_cm, width_cm, height_cm):
    """
    Descarga una imagen de una URL y la dibuja en una posición absoluta
    usando coordenadas FPDF-style (desde arriba).
    """
    try:
        print(f"Intentando descargar: {url}")
        response = requests.get(url, timeout=10)
        
        if response.status_code != 200:
            print(f"Error de servidor: Código {response.status_code}")
            return

        img_data = BytesIO(response.content)

        img_reader = ImageReader(img_data)
        # Verificamos si el buffer tiene contenido
        img_size = len(response.content)
        print(f"Imagen descargada con éxito. Tamaño: {img_size} bytes")

        if img_size < 100: # Si es muy pequeña, algo salió mal
            print("La imagen parece estar vacía o corrupta.")
            return

        # COORDENADAS: Asegúrate de que page_height sea correcto
        page_height = 29.7 * cm 
        y_reportlab = page_height - (y_cm * cm) - (height_cm * cm)
        
        canvas.drawImage(img_reader, x_cm * cm, y_reportlab, 
                         width=width_cm * cm, height=height_cm * cm, 
                         mask='auto')
        print(f"Dibujado en X:{x_cm} Y:{y_cm}")

    except Exception as e:
        print(f"ERROR CRÍTICO: {str(e)}")

def get_imedic_path(key, full_imedic_data):
    """
    Optimización funcional para localizar rutas de imágenes médicas.
    """
    # 1. Mapa de tipos (Diccionario para búsqueda O(1))
    tipos = {
        'colposcopia': 1, 'schiller': 2, 'frotis': 3, 'pap': 4,
        'eco_utero': 5, 'eco_ovarios': 6, 'eco_mamas': 7
    }
    
    ruta_default = '/var/www/docma/img/no-image.jpg'
    tipo_id = tipos.get(key)
    
    if not tipo_id:
        return ruta_default

    # 2. Búsqueda en el dataset coaxial (Ya unificado por full_trace)
    # Buscamos el registro que coincida con el tipo
    registro = next((item for item in full_imedic_data 
                    if item['imedic_tipo_id'] == tipo_id), None)

    if not registro:
        return ruta_default

    # 3. Obtener AA y MM de la visita específica de la imagen
    # En lugar de consultar visitasModel cada vez, hacemos una sola query rápida
    # res[0] es aa, res[1] es mm
    archivo_ruta = f"/var/www/docma/{registro['ruta_local']}/{registro['image_hash']}"
    
    # /var/www/docma/datafiles/visitas/2026/02/06/291602/66bb06c4a36bce55b6f40fe4943fabf0de86e675

    # 4. Verificación física del archivo
    if os.path.exists(archivo_ruta):
        url_image = archivo_ruta
        return url_image

def get_imagenes_adicionales(imedic_data):
    """
    Filtra la lista de imágenes para obtener solo los excedentes.
    Si un tipo de imagen (imedic_tipo_id) ya apareció, las siguientes
    del mismo tipo se consideran 'adicionales'.
    """
    if not imedic_data:
        return []

    buffer_adicionales = []
    stack_tipos_vistos = set()

    for row in imedic_data:
        # Verificamos que tenga un tipo asignado
        tipo_id = row.get('imedic_tipo_id')
        
        if tipo_id in stack_tipos_vistos:
            # Si el tipo ya existe en el reporte principal, 
            # esta imagen es para el "shift" (Página 6)
            buffer_adicionales.append(row)
        else:
            # Es la primera vez que vemos este tipo, 
            # lo marcamos para que no entre al buffer
            stack_tipos_vistos.add(tipo_id)
            
    return buffer_adicionales

def draw_imagenes_adicionales(canvas, fotos_extra):
    """
    Dibuja las imágenes excedentes en páginas nuevas (Página 6 y sucesivas).
    Mantiene la cuadrícula de 2x3 (6 imágenes por página).
    """
    if not fotos_extra:
        return
    
    counter = 0
    ancho_img = 8.0  # cm (80mm en tu PHP)
    alto_img = 6.0   # cm (60mm en tu PHP)
    
    for img in fotos_extra:
        # 1. ¿Necesitamos una nueva página? (Cada 6 imágenes)
        if counter % 6 == 0:
            canvas.showPage()
            # Dibujamos el fondo de la hoja 06
            counter_in_page = 0 # Reiniciamos contador interno de página
            # 2. Calcular posición X (Alternar entre 20mm y 110mm)
            # Si es impar (dentro de la página), columna 1. Si es par, columna 2.
            if (counter_in_page % 2) == 0:
                pos_x = 2.0  # cm (equivale a 20mm)
            else:
                pos_x = 11.0 # cm (equivale a 110mm)
            # 3. Calcular posición Y (40mm, 110mm o 180mm desde arriba)
            fila = counter_in_page // 2
            pos_y = 4.0 + (fila * 7.0) # Cada fila baja 70mm (7cm)            
            # 4. Obtener la ruta del archivo y dibujar
            # Usamos la función optimizada que ya conoce el JOIN de aa/mm/dd
            # Nota: Aquí usamos la data que ya viene en el objeto 'img'
            archivo_ruta = f"/var/www/docma/{img['ruta_local']}/{img['image_hash']}"                
            draw_local_image_absolute(canvas, archivo_ruta, pos_x, pos_y, ancho_img, alto_img)            
            counter += 1
            counter_in_page += 1

def get_dynamic_style(text, base_style):
    """
    Replica la lógica de PHP: encoge la fuente si el texto es muy largo
    para que quepa en el espacio asignado.
    """
    length = len(str(text))
    new_style = ParagraphStyle('Dynamic', parent=base_style)
    
    if length > 500:
        new_style.fontSize = 6
        new_style.leading = 7 # Espacio entre líneas
    elif length > 350:
        new_style.fontSize = 8
        new_style.leading = 9
    else:
        new_style.fontSize = 10
        new_style.leading = 12
        
    return new_style
