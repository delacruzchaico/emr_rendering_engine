def get_flag_values(emr_key, val_selected, def_value=''):
    """
    Traduce la máscara de bits de la tabla maestra.
    - row_maestra: El registro de la tabla maestra (diccionario).
    - val_selected: El valor numérico guardado en la ficha médica (int).
    """
    row_maestra = get_emr_row(emr_key)
    if not val_selected or val_selected == '0':
        return def_value

    decval = int(val_selected)
    data = []

    # Iteramos sobre las llaves del diccionario de la tabla maestra
    for key, value in row_maestra.items():
        # Verificamos si la columna empieza con 'flag' y tiene contenido
        if key.startswith('flag') and value:
            try:
                # Extraemos el número del nombre de la columna (ej: 'flag8' -> 8)
                base = int(key.replace('flag', ''))
                
                # Aplicamos la operación Bitwise AND
                if base & decval:
                    data.append(str(value))
            except ValueError:
                continue # Por si hay un 'flag_algo' que no sea número

    # Retornamos el join con coma o el valor por defecto
    return ", ".join(data) if data else def_value

def get_flag_dataset(emr_key, val_selected):
    """
    Traduce la función dataset($emr_key, $val_selected).
    Retorna una lista de diccionarios con la información de cada flag y su estado.
    """
    # 1. Obtenemos la fila maestra de la configuración (emr_config o similar)
    # Suponiendo que tienes una función genérica para leer las tablas maestras:
    row = get_emr_row( emr_key)
    
    if not row:
        return []

    dec_val = int(val_selected or 0)
    data = []

    # 2. Iteramos sobre las llaves del diccionario (como array_keys)
    for key, value in row.items():
        # Filtramos las que empiezan con 'flag' y tienen contenido
        if key.startswith('flag') and value:
            try:
                # Extraemos el peso del bit (ej. 'flag1' -> 1, 'flag8' -> 8)
                base = int(key.replace('flag', ''))
                
                # Operación bitwise AND
                is_checked = 1 if (base & dec_val) else 0
                
                data.append({
                    'flag': value,
                    'base': base,
                    'checked': is_checked,
                    'selected': 'selected' if is_checked else ''
                })
            except ValueError:
                continue # Por si hay un 'flag_algo' que no termina en número

    # Opcional: Ordenar por base para que salgan en orden (1, 2, 4, 8...)
    data.sort(key=lambda x: x['base'])
    
    return data

def get_visita_data(visita_id):
    query = text("""
    SELECT visitas.*,
    (SELECT clase_atencion_id FROM tipo_atencion WHERE id=visitas.tipo_atencion_id) AS clase_atencion_id,
    (SELECT documento FROM tipo_doc WHERE id=tipo_doc_id) AS tipo_documento,
    nombres,ape_pat,ape_mat,fono1,fono2,celular,nro_doc,pareja_nombres,pareja_apellidos,direccion,fec_nac,info_pendiente,
    TIMESTAMPDIFF(MONTH,fecha,CURDATE()) AS mes_diff,
    TIMESTAMPDIFF(YEAR,fec_nac,NOW()) AS edad,
    LEFT(hora_cita,5) AS hora_cita_format,
    (SELECT motivo FROM motivo_cancela_cita WHERE id=motivo_cancela_cita_id) AS motivo_cancela,
    (SELECT tipo_atencion_id FROM v121_subcita WHERE bestado='1' AND visita_id=visitas.id) AS subcita_id,
    (SELECT titulo FROM medio_conversion WHERE id=medio_conversion_id) AS medio_conversion_titulo,
    (SELECT alias_corporativo FROM sedes WHERE id=sede_id) AS alias_corporativo,
    IFNULL((SELECT MAX(nro_hc) FROM historias AS h WHERE h.bestado='1' AND h.sede_id=visitas.sede_id AND h.paciente_id=visitas.paciente_id),'no asignada') AS nro_hc,
    IFNULL((SELECT MAX(id) FROM historias AS h WHERE h.bestado='1' AND h.sede_id=visitas.sede_id AND h.paciente_id=visitas.paciente_id),'no asignada') AS historia_id,
    (SELECT alias FROM users WHERE id=visitas.insert_user_id) AS insert_alias,
    (SELECT alias FROM users WHERE id=visitas.update_user_id) AS update_alias,
    (SELECT CONCAT(nombres,' ',ape_pat) FROM users WHERE id=medico_id) AS medico,
    (SELECT cargos.nombre FROM users INNER JOIN cargos ON users.cargo_id=cargos.id WHERE users.id=medico_id) AS cargo_medico,
    (SELECT atencion FROM tipo_atencion WHERE id=tipo_atencion_id) AS atencion,
    (SELECT MAX(nro_pap)+1 FROM visitas_pap WHERE bestado='1' AND sede_id=visitas.sede_id) AS next_pap,
    (SELECT 1 FROM v121_telesalud WHERE bestado='1' AND visita_id=visitas.id) AS telesalud,
    IFNULL((SELECT GROUP_CONCAT(nro_pap) FROM visitas_pap WHERE bestado='1' AND visita_id=visitas.id),'') AS gc_nro_pap     
    FROM visitas LEFT JOIN pacientes ON visitas.paciente_id=pacientes.id WHERE visitas.bestado='1' AND visitas.id= :id
    """)
    
    with engine.connect() as conn:
        result = conn.execute(query, {"id": visita_id}).mappings().first()
        return dict(result) if result else None

def filter_visita_pap( visita_id):
    """
    Equivalente a function filter_visita($visita_id) en PHP.
    Busca los resultados de citología asociados a la visita.
    """
    sql = text("""
        SELECT *,
            (SELECT alterado FROM citologia_resultado 
             WHERE id = citologia_resultado_id) AS alterado,
            (SELECT resultado_informe FROM citologia_resultado 
             WHERE id = citologia_resultado_id) AS citologia_resultado
        FROM visitas_pap 
        WHERE bestado = '1' AND visita_id = :visita_id
    """)

    with engine.connect() as conn:
        # Usamos .mappings().all() porque el PHP original devuelve un array (this->query)
        result = conn.execute(sql, {"visita_id": int(visita_id)}).mappings().all()
        
        # Convertimos cada fila a un diccionario real de Python
        return [dict(row) for row in result]

def get_informe_data(visita_id):
    """
      Equivalente a function r($id) en PHP.
      Obtiene toda la data de la visita, paciente, sede y médico en una sola consulta.
    """
    sql = text("""
    SELECT 
    visitas.*,
    atencion,
    LEFT(hora_cita, 5) AS hora_cita,
    (SELECT tagname FROM sedes WHERE id = sede_id) AS sede_tagname,
    (SELECT direccion FROM sedes WHERE id = sede_id) AS sede_direccion,
    (SELECT telefono FROM sedes WHERE id = sede_id) AS sede_telefono,
    (SELECT CONCAT(nombres, ' ', ape_pat, ' ', ape_mat) FROM users WHERE id = medico_id) AS medico,
    (SELECT CONCAT(nombres, ' ', ape_pat, ' ', ape_mat) FROM pacientes WHERE id = paciente_id) AS paciente,
    (SELECT nombres FROM pacientes WHERE id = paciente_id) AS paciente_nombres,
    (SELECT CONCAT(fono1, ' / ', fono2, ' / ', celular) FROM pacientes WHERE id = paciente_id) AS paciente_telefonos,
    (SELECT correo FROM pacientes WHERE id = paciente_id) AS correo,
    (SELECT celular FROM pacientes WHERE id = paciente_id) AS celular,
    
    # Lógica del estado traducida de los IFs de MySQL
    IF(fecha_entrega > '', 'Entregado', 
    IF(fecha_completo > '', 'Completado', 'En espera')
    ) AS estado_entrega,
    
    fecha_aprobado, info_aprobado,
    fecha_completo, v121_informes.info_completo, completo_user_id, aprobado_user_id,
    pdf_last_request, fecha_entrega, info_entrega,
    envio_time, info_envio, envio_user_id,
    cervix_info, frotis_info, colposcopia_info, pap_info, proxima_cita
    
    FROM visitas 
    INNER JOIN tipo_atencion ON visitas.tipo_atencion_id = tipo_atencion.id
    LEFT JOIN v121_informes ON visitas.id = v121_informes.visita_id
    WHERE visitas.bestado = '1' AND visitas.id = :visita_id
    """)
    
    with engine.connect() as conn:
        # Ejecutamos y convertimos a diccionario con .mappings()
        result = conn.execute(sql, {"visita_id": int(visita_id)}).mappings().first()
        
        # Si no hay resultado, devolvemos un dict vacío para evitar el AttributeError: 'NoneType'
        return dict(result) if result else {}

    Trae el listado maestro de diagnósticos (sección 1) y

    """)

    with engine.connect() as conn:
        result = conn.execute(sql, {"visita_id": int(visita_id)}).mappings().all()

def get_masas_lista(visita_id, ubicacion=''):
    """
    Traduce la función lista($visita_id, $ubicacion).
    Obtiene las masas/hallazgos y su descripción CIE-10.

    sql = text(f"""
        SELECT 
            vum.*,
            REPLACE(vum.ubicacion, '_', ' ') AS ubicacion_limpia,
            (SELECT dec10 FROM cie10 WHERE id10 = vum.cie10) AS dec10
        FROM visitas_utero_masas vum
        WHERE vum.bestado = '1' 
          {where_ubicacion} 
          AND vum.visita_id = :visita_id
    """)

    with engine.connect() as conn:
        result = conn.execute(sql, {"visita_id": int(visita_id)}).mappings().all()
        return [dict(r) for r in result]

def get_visitas_coaxiales(visita_id):
    # Usamos COALESCE o una lógica que maneje si no hay epago_id
    sql_coaxial = text("""
        SELECT :v_id AS visita_id UNION 
        SELECT visita_id 
        FROM epagos_visitas 
        WHERE bestado='1' AND epago_id = (SELECT epago_id FROM epagos_visitas WHERE visita_id = :v_id  LIMIT 1)

    """)

    try:
        with engine.connect() as conn:
            # SQLAlchemy 2.0 requiere explícitamente el commit si hay cambios, 
            # pero para SELECT solo fetchall()
            resultados = conn.execute(sql_coaxial, {"v_id": visita_id}).fetchall()

            # Retornamos lista simple de IDs, filtrando posibles None
            return [r[0] for r in resultados if r[0] is not None]
    except Exception as e:
        # Como arquitecto, siempre loguea el error antes de fallar
        print(f"Error en get_visitas_coaxiales: {e}")
        return [visita_id] # Por seguridad, retorna al menos la visita actual

def get_lista_imedic(visita_id):
    sql = text("""SELECT *,
    (SELECT CONCAT(nombres,' ',ape_pat) FROM users WHERE id=medico_id) AS medico_image,
    visitas.fecha AS fecha_image,
    (SELECT tagname FROM imedic_tipo WHERE id=imedic_tipo_id) AS imedic_tipo_tagname,
    CASE WHEN visitas_archives.filetype LIKE '%/png' THEN 'png' WHEN filetype LIKE '%/jp_g' THEN 'jpg' ELSE 'jpg' END filetype_short,
    CONCAT('datafiles/visitas/',YEAR(visitas.insert_time) ,'/',LPAD(MONTH(visitas.insert_time),2,'0'),'/',LPAD(DAY(visitas.insert_time),2,'0'),'/',visitas.id) AS ruta_local,
    visitas_imedic.insert_time,
    visitas_imedic.id
    FROM visitas_imedic 
    INNER JOIN docma_archives.visitas_archives 
    ON visitas_imedic.visita_id=visitas_archives.visita_id 
    AND visitas_imedic.image_hash=visitas_archives.filesha1
    INNER JOIN visitas ON visitas.id=visitas_imedic.visita_id
    WHERE visitas_archives.bestado='1' AND visitas_imedic.bestado='1' AND visitas_imedic.visita_id= :v_id 
    ORDER BY imedic_tipo_id ASC,visitas_imedic.insert_time ASC""")

    with engine.connect() as conn:
        result = conn.execute(sql, {"v_id": visita_id})
        # Convertimos a lista de diccionarios (mapeando nombres de columnas)
        return [dict(row._mapping) for row in result]

    ids_coaxiales = get_visitas_coaxiales( visita_id)
